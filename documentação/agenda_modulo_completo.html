Módulo de Agendamento
para seu CRM SaaS

Sincronização bidirecional com Google Calendar, construída sobre o seu schema Supabase existente. O banco de dados é a fonte da verdade — o Google Calendar é apenas o espelho.

Supabase + PostgreSQL Google Calendar API OAuth 2.0 RLS por organization_id Triggers de Auditoria BullMQ / Redis FullCalendar.io
STEP 00

Arquitetura Geral

Como as peças se encaixam. Entenda antes de escrever uma linha de código.

Princípio Central

🏛️
O banco do CRM é a lei. O Google Calendar é o tradutor.

Toda operação começa e termina no PostgreSQL. O Google Calendar recebe uma cópia sincronizada, mas nunca dita o estado do sistema. Se o Google cair, seu CRM continua funcionando normalmente.

Como as novas tabelas se integram com o que você já tem.

✅
Seu schema já tem a base certa

O padrão organization_id como tenant, profiles como usuário e organization_members para permissões são exatamente o que precisamos. Não há conflitos.

ENUMs

SQL — ENUMs 
-- Status do agendamento
DO $$ BEGIN
  CREATE TYPE appointment_status_enum AS ENUM (
    'scheduled', 'confirmed', 'cancelled', 'completed', 'no_show'
  );
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

-- Status de sincronização
DO $$ BEGIN
  CREATE TYPE sync_status_enum AS ENUM (
    'pending', 'synced', 'failed', 'conflict'
  );
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

-- Direção da sync
DO $$ BEGIN
  CREATE TYPE sync_direction_enum AS ENUM (
    'crm_to_google', 'google_to_crm'
  );
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

-- Ação de sync
DO $$ BEGIN
  CREATE TYPE sync_action_enum AS ENUM (
    'create', 'update', 'delete', 'conflict'
  );
EXCEPTION WHEN duplicate_object THEN NULL; END $$;

Tabela: appointments

SQL — appointments 
CREATE TABLE IF NOT EXISTS appointments (
  id                  UUID        PRIMARY KEY DEFAULT gen_random_uuid(),

  -- Relacionamentos com seu schema existente
  organization_id     UUID        NOT NULL
                      REFERENCES organizations(id) ON DELETE CASCADE,
  lead_id             UUID        NULL
                      REFERENCES leads(id) ON DELETE SET NULL,
  user_id             UUID        NULL
                      REFERENCES profiles(id) ON DELETE SET NULL,
  created_by          UUID        NOT NULL
                      REFERENCES profiles(id) ON DELETE SET NULL,

  -- Dados do agendamento
  title               TEXT        NOT NULL,
  description         TEXT        NULL,
  location            TEXT        NULL,
  start_at            TIMESTAMPTZ NOT NULL,
  end_at              TIMESTAMPTZ NOT NULL,
  timezone            TEXT        NOT NULL DEFAULT 'UTC',
  status              appointment_status_enum NOT NULL DEFAULT 'scheduled',

  -- Recorrência (RFC 5545 RRULE)
  recurrence_rule     TEXT        NULL,
  recurrence_parent   UUID        NULL
                      REFERENCES appointments(id) ON DELETE CASCADE,

  -- Participantes externos
  attendees           JSONB       NOT NULL DEFAULT '[]',
  metadata            JSONB       NOT NULL DEFAULT '{}'{,

  -- Google Calendar
  google_event_id     TEXT        NULL,
  google_etag         TEXT        NULL,
  google_html_link    TEXT        NULL,
  sync_status         sync_status_enum NOT NULL DEFAULT 'pending',
  last_synced_at      TIMESTAMPTZ NULL,
  sync_error          TEXT        NULL,

  -- Versionamento para resolução de conflitos
  version             INTEGER     NOT NULL DEFAULT 1,
  source              TEXT        NOT NULL DEFAULT 'crm'
                      CHECK (source IN ('crm', 'google', 'api', 'agent')),

  -- Soft delete
  created_at          TIMESTAMPTZ NOT NULL DEFAULT timezone('utc', now()),
  updated_at          TIMESTAMPTZ NOT NULL DEFAULT now(),
  deleted_at          TIMESTAMPTZ NULL,

  CONSTRAINT appt_end_after_start CHECK (end_at > start_at),
  CONSTRAINT appt_title_not_empty CHECK (trim(title) <> '')
);

CREATE TABLE IF NOT EXISTS calendar_integrations (

  created_at            TIMESTAMPTZ NOT NULL DEFAULT timezone('utc', now()),

  UNIQUE (user_id, organization_id, provider, provider_account_id)

STEP 03

Políticas RLS (Row Level Security)

Isolamento total entre organizações. Usa organization_members como guardião.

⚠️
Service Role Key no backend

Workers e webhooks usam SUPABASE_SERVICE_ROLE_KEY, que bypassa RLS. Nunca exponha essa chave no frontend. O cliente usa apenas a ANON_KEY.

SQL — Funções Helper + RLS Completo 
-- Retorna organization_ids do usuário logado (reutilizável em todas as policies)
CREATE OR REPLACE FUNCTION get_my_org_ids()
RETURNS SETOF UUID LANGUAGE sql STABLE SECURITY DEFINER AS $$
  SELECT organization_id FROM organization_members WHERE user_id = auth.uid();
$$;

-- Verifica se usuário tem um role específico na org
CREATE OR REPLACE FUNCTION has_org_role(org_id UUID, required_role TEXT)
RETURNS BOOLEAN LANGUAGE sql STABLE SECURITY DEFINER AS $$
  SELECT EXISTS (
    SELECT 1 FROM organization_members
    WHERE user_id = auth.uid() AND organization_id = org_id AND role = required_role
  );
$$;

-- ── APPOINTMENTS ──
ALTER TABLE appointments ENABLE ROW LEVEL SECURITY;

CREATE POLICY "appointments: membros leem" ON appointments FOR SELECT
  USING (organization_id IN (SELECT get_my_org_ids()));

CREATE POLICY "appointments: membros criam" ON appointments FOR INSERT
  WITH CHECK (organization_id IN (SELECT get_my_org_ids()) AND created_by = auth.uid());

CREATE POLICY "appointments: membros editam" ON appointments FOR UPDATE
  USING (organization_id IN (SELECT get_my_org_ids()))
  WITH CHECK (organization_id IN (SELECT get_my_org_ids()));

-- ── CALENDAR_INTEGRATIONS ──
ALTER TABLE calendar_integrations ENABLE ROW LEVEL SECURITY;

CREATE POLICY "calint: ver próprias" ON calendar_integrations FOR SELECT
  USING (user_id = auth.uid());

CREATE POLICY "calint: criar própria" ON calendar_integrations FOR INSERT
  WITH CHECK (user_id = auth.uid() AND organization_id IN (SELECT get_my_org_ids()));

CREATE POLICY "calint: editar própria" ON calendar_integrations FOR UPDATE
  USING (user_id = auth.uid()) WITH CHECK (user_id = auth.uid());

CREATE POLICY "calint: desconectar" ON calendar_integrations FOR DELETE
  USING (user_id = auth.uid());

-- ── SYNC_LOGS ──
ALTER TABLE sync_logs ENABLE ROW LEVEL SECURITY;

CREATE POLICY "synclogs: membros leem" ON sync_logs FOR SELECT
  USING (organization_id IN (SELECT get_my_org_ids()));

CREATE POLICY "synclogs: service role insere" ON sync_logs FOR INSERT
  WITH CHECK (auth.role() = 'service_role');

-- ── AUDIT_LOGS ──
ALTER TABLE audit_logs ENABLE ROW LEVEL SECURITY;

CREATE POLICY "auditlogs: admin lê" ON audit_logs FOR SELECT
  USING (organization_id IN (SELECT get_my_org_ids())
    AND has_org_role(organization_id, 'admin'));

CREATE POLICY "auditlogs: service role insere" ON audit_logs FOR INSERT
  WITH CHECK (auth.role() = 'service_role');

-- ── INTERNAL_WEBHOOKS ──
ALTER TABLE internal_webhooks ENABLE ROW LEVEL SECURITY;

CREATE POLICY "webhooks: membros leem" ON internal_webhooks FOR SELECT
  USING (organization_id IN (SELECT get_my_org_ids()));

CREATE POLICY "webhooks: admin gerencia" ON internal_webhooks FOR ALL
  USING (has_org_role(organization_id, 'admin'))
  WITH CHECK (has_org_role(organization_id, 'admin'));
STEP 04

Triggers de Auditoria

Automação no banco: auditoria, versionamento, updated_at e integração com activities.

SQL — Função de Auditoria Genérica 
CREATE OR REPLACE FUNCTION fn_audit_log()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
  v_user_id   UUID;
  v_org_id    UUID;
  v_old_data  JSONB;
  v_new_data  JSONB;
BEGIN
  BEGIN
    v_user_id := auth.uid();
  EXCEPTION WHEN OTHERS THEN
    v_user_id := NULL;
  END;

  IF TG_OP = 'DELETE' THEN
    BEGIN v_org_id := OLD.organization_id; EXCEPTION WHEN OTHERS THEN v_org_id := NULL; END;
    v_old_data := to_jsonb(OLD); v_new_data := NULL;
  ELSIF TG_OP = 'INSERT' THEN
    BEGIN v_org_id := NEW.organization_id; EXCEPTION WHEN OTHERS THEN v_org_id := NULL; END;
    v_old_data := NULL; v_new_data := to_jsonb(NEW);
  ELSE
    BEGIN v_org_id := NEW.organization_id; EXCEPTION WHEN OTHERS THEN v_org_id := NULL; END;
    v_old_data := to_jsonb(OLD); v_new_data := to_jsonb(NEW);
  END IF;

  -- Remove tokens da auditoria mesmo que estejam criptografados
  IF TG_TABLE_NAME = 'calendar_integrations' THEN
    v_old_data := v_old_data - 'access_token_enc' - 'refresh_token_enc';
    v_new_data := v_new_data - 'access_token_enc' - 'refresh_token_enc';
  END IF;

  INSERT INTO audit_logs (user_id, organization_id, entity_type, entity_id, action, old_data, new_data, source)
  VALUES (v_user_id, v_org_id, TG_TABLE_NAME,
    COALESCE(CASE WHEN TG_OP = 'DELETE' THEN OLD.id ELSE NEW.id END, gen_random_uuid()),
    TG_OP, v_old_data, v_new_data, 'trigger');

  RETURN COALESCE(NEW, OLD);
END;
$$;

-- Aplica auditoria nas tabelas
DROP TRIGGER IF EXISTS trg_appointments_audit ON appointments;
CREATE TRIGGER trg_appointments_audit
  AFTER INSERT OR UPDATE OR DELETE ON appointments
  FOR EACH ROW EXECUTE FUNCTION fn_audit_log();

-- Incrementa version apenas em mudanças de dados reais
CREATE OR REPLACE FUNCTION fn_increment_appointment_version()
RETURNS TRIGGER LANGUAGE plpgsql AS $$
BEGIN
  IF (OLD.title IS DISTINCT FROM NEW.title OR OLD.start_at IS DISTINCT FROM NEW.start_at
    OR OLD.end_at IS DISTINCT FROM NEW.end_at OR OLD.status IS DISTINCT FROM NEW.status
    OR OLD.attendees IS DISTINCT FROM NEW.attendees) THEN
    NEW.version := OLD.version + 1;
  END IF;
  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS trg_appointments_version ON appointments;
CREATE TRIGGER trg_appointments_version
  BEFORE UPDATE ON appointments FOR EACH ROW EXECUTE FUNCTION fn_increment_appointment_version();

-- Popula sua tabela activities automaticamente ao criar/editar appointments
CREATE OR REPLACE FUNCTION fn_appointment_to_activity()
RETURNS TRIGGER LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE
  v_desc  TEXT;
  v_type  TEXT;
BEGIN
  IF NEW.lead_id IS NULL THEN RETURN NEW; END IF;

  IF TG_OP = 'INSERT' THEN
    v_type := 'appointment_created';
    v_desc := format('Agendamento criado: "%s" em %s', NEW.title,
      to_char(NEW.start_at AT TIME ZONE NEW.timezone, 'DD/MM/YYYY HH24:MI'));
  ELSIF OLD.status IS DISTINCT FROM NEW.status THEN
    v_type := 'appointment_status_changed';
    v_desc := format('Status alterado: %s → %s ("%s")', OLD.status, NEW.status, NEW.title);
  ELSIF OLD.start_at IS DISTINCT FROM NEW.start_at THEN
    v_type := 'appointment_rescheduled';
    v_desc := format('Reagendado para %s: "%s"',
      to_char(NEW.start_at AT TIME ZONE NEW.timezone, 'DD/MM/YYYY HH24:MI'), NEW.title);
  ELSE
    RETURN NEW;
  END IF;

  INSERT INTO activities (lead_id, organization_id, activity_type, description)
  VALUES (NEW.lead_id, NEW.organization_id, v_type, v_desc);

  RETURN NEW;
END;
$$;

DROP TRIGGER IF EXISTS trg_appointment_activity ON appointments;
CREATE TRIGGER trg_appointment_activity
  AFTER INSERT OR UPDATE ON appointments
  FOR EACH ROW EXECUTE FUNCTION fn_appointment_to_activity();

Funções RPC (Supabase)

Funções chamáveis via supabase.rpc() do frontend ou backend.

SQL — check_appointment_conflicts 
-- Verifica sobreposição de horários. Chamar antes de criar/editar appointment.
CREATE OR REPLACE FUNCTION check_appointment_conflicts(
  p_user_id         UUID,
  p_organization_id UUID,
  p_start_at        TIMESTAMPTZ,
  p_end_at          TIMESTAMPTZ,
  p_exclude_id      UUID DEFAULT NULL
)
RETURNS TABLE(id UUID, title TEXT, start_at TIMESTAMPTZ, end_at TIMESTAMPTZ, status appointment_status_enum)
LANGUAGE sql STABLE SECURITY DEFINER AS $$
  SELECT a.id, a.title, a.start_at, a.end_at, a.status
  FROM appointments a
  WHERE a.user_id = p_user_id
    AND a.organization_id = p_organization_id
    AND a.deleted_at IS NULL
    AND a.status NOT IN ('cancelled', 'completed', 'no_show')
    AND a.id IS DISTINCT FROM p_exclude_id
    AND a.start_at < p_end_at
    AND a.end_at   > p_start_at;
$$;
SQL — get_available_slots + soft_delete_appointment 
-- Retorna slots livres de um usuário. Usado pelo endpoint de IA e pela UI.
CREATE OR REPLACE FUNCTION get_available_slots(
  p_user_id         UUID,
  p_organization_id UUID,
  p_date            DATE,
  p_duration_min    INTEGER DEFAULT 60,
  p_work_start_hour INTEGER DEFAULT 8,
  p_work_end_hour   INTEGER DEFAULT 18,
  p_timezone        TEXT    DEFAULT 'America/Sao_Paulo'
)
RETURNS TABLE(slot_start TIMESTAMPTZ, slot_end TIMESTAMPTZ)
LANGUAGE plpgsql STABLE SECURITY DEFINER AS $$
DECLARE
  v_work_start  TIMESTAMPTZ;
  v_work_end    TIMESTAMPTZ;
  v_cursor      TIMESTAMPTZ;
  v_slot_dur    INTERVAL := (p_duration_min || ' minutes')::INTERVAL;
BEGIN
  v_work_start := (p_date::TEXT || ' ' || p_work_start_hour::TEXT || ':00:00')::TIMESTAMP
                   AT TIME ZONE p_timezone;
  v_work_end   := (p_date::TEXT || ' ' || p_work_end_hour::TEXT || ':00:00')::TIMESTAMP
                   AT TIME ZONE p_timezone;
  v_cursor := v_work_start;

  WHILE v_cursor + v_slot_dur <= v_work_end LOOP
    IF NOT EXISTS (
      SELECT 1 FROM appointments a
      WHERE a.user_id = p_user_id AND a.organization_id = p_organization_id
        AND a.deleted_at IS NULL AND a.status NOT IN ('cancelled', 'completed', 'no_show')
        AND a.start_at < v_cursor + v_slot_dur AND a.end_at > v_cursor
    ) THEN
      slot_start := v_cursor; slot_end := v_cursor + v_slot_dur;
      RETURN NEXT;
    END IF;
    v_cursor := v_cursor + v_slot_dur;
  END LOOP;
END;
$$;

-- Soft delete seguro: marca deleted_at, agenda sync com Google para deletar lá também
CREATE OR REPLACE FUNCTION soft_delete_appointment(
  p_appointment_id  UUID,
  p_user_id         UUID,
  p_organization_id UUID
)
RETURNS BOOLEAN LANGUAGE plpgsql SECURITY DEFINER AS $$
DECLARE v_rows INTEGER;
BEGIN
  UPDATE appointments SET
    deleted_at  = timezone('utc', now()),
    status      = 'cancelled',
    sync_status = 'pending'
  WHERE id = p_appointment_id AND organization_id = p_organization_id AND deleted_at IS NULL;
  GET DIAGNOSTICS v_rows = ROW_COUNT;
  RETURN v_rows > 0;
END;
$$;

async function handleGoogleCallback(req, res) {
  const { code, state } = req.query;

  const decoded = jwt.verify(state, process.env.STATE_JWT_SECRET);
  if (!decoded?.user_id) throw new Error('Invalid state');

  const { tokens } = await oauth2Client.getToken(code);
  // tokens = { access_token, refresh_token, expiry_date, id_token }

  const ticket = await oauth2Client.verifyIdToken({ idToken: tokens.id_token });
  const payload = ticket.getPayload();
  // payload.sub = Google account ID único do usuário

  await saveEncryptedTokens(decoded.user_id, decoded.org_id, payload.sub, tokens);
  await setupGoogleWatch(decoded.user_id, decoded.org_id);

  res.redirect('/settings/integrations?connected=true');
}

AES-256-GCM. Um vazamento do banco não expõe os tokens.

🔐
NUNCA salve tokens em texto puro

Os tokens dão acesso total à agenda do usuário. Use criptografia antes de persistir. A chave de criptografia nunca vai para o banco — fica somente nas variáveis de ambiente do servidor.

JavaScript — Criptografia AES-256-GCM + Salvar no Supabase 
const crypto = require('crypto');

// TOKEN_ENCRYPTION_KEY = 64 hex chars = 32 bytes = AES-256
// Gere com: node -e "console.log(crypto.randomBytes(32).toString('hex'))"
const ENC_KEY = Buffer.from(process.env.TOKEN_ENCRYPTION_KEY, 'hex');

function encrypt(text) {
  const iv     = crypto.randomBytes(12);
  const cipher = crypto.createCipheriv('aes-256-gcm', ENC_KEY, iv);
  const enc    = Buffer.concat([cipher.update(text, 'utf8'), cipher.final()]);
  const tag    = cipher.getAuthTag();
  // Formato: base64(iv[12] + tag[16] + ciphertext)
  return Buffer.concat([iv, tag, enc]).toString('base64');
}

function decrypt(b64) {
  const buf     = Buffer.from(b64, 'base64');
  const iv      = buf.slice(0, 12);
  const tag     = buf.slice(12, 28);
  const enc     = buf.slice(28);
  const decipher = crypto.createDecipheriv('aes-256-gcm', ENC_KEY, iv);
  decipher.setAuthTag(tag);
  return decipher.update(enc).toString('utf8') + decipher.final('utf8');
}

async function saveEncryptedTokens(userId, orgId, providerAccountId, tokens) {
  const { error } = await serviceClient
    .from('calendar_integrations')
    .upsert({
      user_id:              userId,
      organization_id:      orgId,
      provider_account_id:  providerAccountId,
      access_token_enc:     encrypt(tokens.access_token),
      refresh_token_enc:    tokens.refresh_token ? encrypt(tokens.refresh_token) : undefined,
      token_expires_at:     new Date(tokens.expiry_date).toISOString(),
      scopes:               tokens.scope?.split(' ') ?? [],
    }, { onConflict: 'user_id,organization_id,provider,provider_account_id' });

  if (error) throw error;
}

// Renovação automática — chame antes de cada request ao Google
async function getValidAccessToken(userId, orgId) {
  const { data } = await serviceClient
    .from('calendar_integrations')
    .select('*')
    .eq('user_id', userId).eq('organization_id', orgId).eq('is_active', true)
    .single();

  const expiresAt  = new Date(data.token_expires_at);
  const needsRenew = expiresAt - Date.now() < 5 * 60 * 1000; // < 5 min

  if (!needsRenew) return decrypt(data.access_token_enc);

  oauth2Client.setCredentials({ refresh_token: decrypt(data.refresh_token_enc) });
  const { credentials } = await oauth2Client.refreshAccessToken();

  await serviceClient.from('calendar_integrations').update({
    access_token_enc: encrypt(credentials.access_token),
    token_expires_at: new Date(credentials.expiry_date).toISOString()
  }).eq('id', data.id);

  return credentials.access_token;
}

// POST /api/appointments
async function createAppointment(req, res) {
  const { title, start_at, end_at, description, lead_id, attendees } = req.body;
  const { id: userId, org_id: orgId } = req.user;

  // 1. Verifica conflitos via RPC do Supabase
  const { data: conflicts } = await supabase.rpc('check_appointment_conflicts', {
    p_user_id: userId, p_organization_id: orgId, p_start_at: start_at, p_end_at: end_at
  });
  if (conflicts?.length) return res.status(409).json({ error: 'TIME_CONFLICT', conflicts });

  // 2. Salva no banco — fonte da verdade
  const { data: appt, error } = await serviceClient
    .from('appointments')
    .insert({
      organization_id: orgId, user_id: userId, created_by: userId,
      title, description, start_at, end_at, lead_id,
      attendees: attendees ?? [],
      sync_status: 'pending', source: 'crm'
    })
    .select().single();

  if (error) throw error;

  // 3. Enfileira sync assíncrono (BullMQ)
  await syncQueue.add('sync-to-google',
    { appointmentId: appt.id, userId, orgId, action: 'create' },
    { attempts: 5, backoff: { type: 'exponential', delay: 2000 } }
  );

  // 4. Responde imediatamente — não espera o Google
  res.status(201).json({ data: appt });
}

// WORKER — processa job da fila
syncQueue.process('sync-to-google', async (job) => {
  const { appointmentId, userId, orgId, action } = job.data;
  const { data: appt } = await serviceClient.from('appointments')
    .select().eq('id', appointmentId).single();

  if (!appt || appt.deleted_at) return; // foi deletado antes do sync

  const accessToken = await getValidAccessToken(userId, orgId);
  const calendar    = google.calendar({ version: 'v3', auth: authClient(accessToken) });

  const googleEvent = {
    summary:     appt.title,
    description: appt.description,
    location:    appt.location,
    start: { dateTime: appt.start_at, timeZone: appt.timezone },
    end:   { dateTime: appt.end_at,   timeZone: appt.timezone },
    attendees: appt.attendees.map(a => ({ email: a.email })),
    extendedProperties: { private: { crm_appointment_id: appt.id } }
  };

  const start = Date.now();
  try {
    let response;
    if (action === 'create') {
      response = await calendar.events.insert({ calendarId: 'primary', resource: googleEvent });
      await serviceClient.from('appointments').update({
        google_event_id: response.data.id, google_etag: response.data.etag,
        sync_status: 'synced', last_synced_at: new Date().toISOString()
      }).eq('id', appointmentId);

    } else if (action === 'update' && appt.google_event_id) {
      response = await calendar.events.update({
        calendarId: 'primary', eventId: appt.google_event_id, resource: googleEvent
      });
      await serviceClient.from('appointments').update({
        google_etag: response.data.etag, sync_status: 'synced', last_synced_at: new Date().toISOString()
      }).eq('id', appointmentId);

    } else if (action === 'delete' && appt.google_event_id) {
      await calendar.events.delete({ calendarId: 'primary', eventId: appt.google_event_id });
    }

    await serviceClient.from('sync_logs').insert({
      appointment_id: appointmentId, user_id: userId, organization_id: orgId,
      direction: 'crm_to_google', action, status: 'success',
      attempt_number: job.attemptsMade + 1, duration_ms: Date.now() - start
    });

  } catch (err) {
    await serviceClient.from('appointments').update({
      sync_status: 'failed', sync_error: err.message
    }).eq('id', appointmentId);
    await serviceClient.from('sync_logs').insert({
      appointment_id: appointmentId, user_id: userId, organization_id: orgId,
      direction: 'crm_to_google', action, status: 'failed',
      error_code: String(err.code), error_message: err.message,
      attempt_number: job.attemptsMade + 1
    });
    throw err; // BullMQ faz retry automático com backoff exponencial
  }
});

// POST /webhooks/google/calendar — endpoint público
async function handleGoogleWebhook(req, res) {
  res.status(200).send(); // Google exige resposta rápida

  const channelId  = req.headers['x-goog-channel-id'];
  const resourceId = req.headers['x-goog-resource-id'];
  const token      = req.headers['x-goog-channel-token'];
  const state      = req.headers['x-goog-resource-state'];
  const msgNumber  = req.headers['x-goog-message-number'];

  if (token !== process.env.WEBHOOK_VERIFICATION_TOKEN) return;
  if (state === 'sync') return; // primeira notificação de handshake

  // Idempotência: ignora mensagens já processadas (proteção contra replay)
  const idempotencyKey = `webhook:${channelId}:${msgNumber}`;
  const already = await redis.get(idempotencyKey);
  if (already) return;
  await redis.set(idempotencyKey, '1', 'EX', 86400);

  const { data: integration } = await serviceClient
    .from('calendar_integrations')
    .select('user_id, organization_id, id')
    .eq('webhook_channel_id', channelId).eq('webhook_resource_id', resourceId)
    .single();

  if (!integration) return;

  await syncQueue.add('sync-from-google', {
    userId:        integration.user_id,
    orgId:         integration.organization_id,
    integrationId: integration.id
  });
}

// WORKER — sincronização incremental do Google para o CRM
syncQueue.process('sync-from-google', async (job) => {
  const { userId, orgId, integrationId } = job.data;
  const { data: integ } = await serviceClient.from('calendar_integrations')
    .select().eq('id', integrationId).single();

  const accessToken = await getValidAccessToken(userId, orgId);
  const calendar    = google.calendar({ version: 'v3', auth: authClient(accessToken) });

  const params = { calendarId: 'primary', singleEvents: true };
  if (integ.sync_token) {
    params.syncToken = integ.sync_token; // incremental
  } else {
    params.timeMin = new Date(Date.now() - 30 * 86400000).toISOString(); // full sync
  }

  try {
    const { data } = await calendar.events.list(params);
    for (const event of data.items ?? []) {
      await upsertEventFromGoogle(userId, orgId, event);
    }
    await serviceClient.from('calendar_integrations').update({
      sync_token: data.nextSyncToken, last_synced_at: new Date().toISOString()
    }).eq('id', integrationId);
  } catch (err) {
    if (err.code === 410) { // syncToken expirou
      await serviceClient.from('calendar_integrations').update({ sync_token: null }).eq('id', integrationId);
    }
    throw err;
  }
});

Upsert e Resolução de Conflitos

Como processar eventos do Google sem criar duplicatas ou perder dados.

JavaScript — upsertEventFromGoogle (Last-Write-Wins) 
async function upsertEventFromGoogle(userId, orgId, googleEvent) {
  // Evento deletado no Google → soft delete no CRM
  if (googleEvent.status === 'cancelled') {
    await serviceClient.from('appointments').update({
      deleted_at: new Date().toISOString(), sync_status: 'synced'
    }).eq('google_event_id', googleEvent.id).eq('organization_id', orgId).is('deleted_at', null);
    return;
  }

  // Verifica se o evento foi criado pelo próprio CRM
  const crmId = googleEvent.extendedProperties?.private?.crm_appointment_id;

  if (crmId) {
    // Evento veio do CRM — verifica se houve mudança real via etag
    const { data: existing } = await serviceClient.from('appointments')
      .select().eq('id', crmId).single();

    if (!existing || existing.google_etag === googleEvent.etag) return; // sem mudança

    // Conflito: evento foi editado diretamente no Google
    const crmUpdatedAt    = new Date(existing.updated_at);
    const googleUpdatedAt = new Date(googleEvent.updated);
    const googleWins      = googleUpdatedAt > crmUpdatedAt;

    await serviceClient.from('sync_logs').insert({
      appointment_id: existing.id, user_id: userId, organization_id: orgId,
      direction: 'google_to_crm', action: 'conflict', status: 'success',
      response_body: { resolution: googleWins ? 'google_wins' : 'crm_wins',
        crm_updated: crmUpdatedAt, google_updated: googleUpdatedAt }
    });

    if (googleWins) {
      await serviceClient.from('appointments').update({
        title:         googleEvent.summary,
        description:   googleEvent.description,
        start_at:      googleEvent.start.dateTime ?? googleEvent.start.date,
        end_at:        googleEvent.end.dateTime   ?? googleEvent.end.date,
        google_etag:   googleEvent.etag,
        sync_status:   'synced', last_synced_at: new Date().toISOString()
      }).eq('id', crmId);
    } else {
      // CRM ganhou: re-enfileira para sobrescrever o Google
      await syncQueue.add('sync-to-google', { appointmentId: crmId, userId, orgId, action: 'update' });
    }
    return;
  }

  // Evento criado diretamente no Google → criar no CRM
  await serviceClient.from('appointments').upsert({
    user_id:          userId,
    organization_id:  orgId,
    created_by:       userId,
    title:            googleEvent.summary ?? '(Sem título)',
    description:      googleEvent.description,
    location:         googleEvent.location,
    start_at:         googleEvent.start.dateTime ?? googleEvent.start.date,
    end_at:           googleEvent.end.dateTime   ?? googleEvent.end.date,
    timezone:         googleEvent.start.timeZone ?? 'UTC',
    google_event_id:  googleEvent.id,
    google_etag:      googleEvent.etag,
    sync_status:      'synced', last_synced_at: new Date().toISOString(),
    source:           'google'
  }, { onConflict: 'google_event_id', ignoreDuplicates: false });
}

import FullCalendar from '@fullcalendar/react';
import dayGridPlugin     from '@fullcalendar/daygrid';

import { useQueryClient } from '@tanstack/react-query';
import { supabase } from './lib/supabase';
import { useEffect } from 'react';

export function AppointmentsCalendar({ orgId, userId }) {
  const queryClient = useQueryClient();

  // Realtime: Supabase notifica mudanças na tabela appointments
  useEffect(() => {
    const channel = supabase
      .channel(`appointments:${orgId}`)
      .on('postgres_changes', {
        event: '*', schema: 'public', table: 'appointments',
        filter: `organization_id=eq.${orgId}`
      }, () => {
        queryClient.invalidateQueries(['appointments']);
      })
      .subscribe();

    return () => supabase.removeChannel(channel);
  }, [orgId]);

  // Busca eventos do Supabase por range de datas
  const fetchEvents = async (fetchInfo) => {
    const { data } = await supabase
      .from('appointments')
      .select('id, title, start_at, end_at, status, sync_status, lead_id')
      .eq('organization_id', orgId)
      .is('deleted_at', null)
      .gte('start_at', fetchInfo.startStr)
      .lte('end_at', fetchInfo.endStr);

    return (data ?? []).map(appt => ({
      id:    appt.id,
      title: appt.title,
      start: appt.start_at,
      end:   appt.end_at,
      backgroundColor: appt.sync_status === 'failed' ? '#ff4d6a'
                     : appt.status === 'confirmed'    ? '#00e5a0' : '#3d7eff',
      extendedProps: appt
    }));
  };

  // Drag & drop — atualiza horário no backend
  const handleEventDrop = async ({ event, revert }) => {
    const { error } = await supabase.from('appointments').update({
      start_at: event.start.toISOString(),
      end_at:   event.end.toISOString(),
      sync_status: 'pending'  // marca para re-sync
    }).eq('id', event.id);

    if (error) { revert(); return; }
    // Dispara job de sync no backend via API
    fetch(`/api/appointments/${event.id}/sync`, { method: 'POST' });
  };

  return (
    <FullCalendar
      plugins={[dayGridPlugin, timeGridPlugin, interactionPlugin]}
      initialView='timeGridWeek'
      headerToolbar={{ left: 'prev,next today', center: 'title',
                       right: 'dayGridMonth,timeGridWeek,timeGridDay' }}
      editable={true}
      selectable={true}
      events={fetchEvents}
      eventDrop={handleEventDrop}
      eventResize={handleEventDrop}
      select={(info) => openCreateModal(info.start, info.end)}
      locale='pt-br'
      slotMinTime='07:00:00'
      slotMaxTime='22:00:00'
      nowIndicator={true}
